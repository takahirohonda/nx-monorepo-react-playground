```ts
// function example.
function speakGerman() {
  const sentence = document.getElementById('sentence')?.textContent
  if (sentence) {
    const utterance = new SpeechSynthesisUtterance()
    utterance.lang = 'de'
    utterance.text = sentence
    speechSynthesis.speak(utterance)
  }
}
```
